    def __getitem__(self, build_id) -> AVBuild:
        for rel in self:
            if rel.buildId == build_id:
                return rel
        raise AttributeError('job not found: {}'.format(build_id))

    def __len__(self) -> int:

    def __contains__(self, build_id) -> bool:

        except AttributeError:
            return False
