    'NOTSET': 0,
    'DEBUG': 10,
    'INFO': 20,
    'WARNING': 30,
    'ERROR': 40,
    'CRITICAL': 50,
}

LOGGING_NAMES = {LOGGING_LEVELS[k]: k for k in LOGGING_LEVELS}


def _sanitize_level(level):
    if isinstance(level, int):
        return level
    elif level in LOGGING_LEVELS:
        return LOGGING_LEVELS[level]
    else:
        raise KeyError(level)
